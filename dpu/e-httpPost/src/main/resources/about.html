The configuration dialog offers possibility to configure only one POST request. In order to execute multiple requests RDF configuration must be used. If the URI in the dialog configuration is left blank, then no POST request based on the dialog is executed.
Configuration sample: 
<http://localhost/resource/config>

    <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <http://unifiedviews.eu/ontology/dpu/httpPost/Configuration>;

    <http://unifiedviews.eu/ontology/dpu/httpPost/requests> <http://localhost/resource/request/0>.



<http://localhost/resource/request/0>

    <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <http://unifiedviews.eu/ontology/dpu/httpPost/Request>;

    <http://unifiedviews.eu/ontology/dpu/httpPost/uri> "http://eagri.cz/public/app/svs_pub/mapy_vk/bin/maps_read_data.php";

    <http://unifiedviews.eu/ontology/dpu/httpPost/fileName> "E01_R2015";

	<http://unifiedviews.eu/ontology/dpu/httpPost/argument> <http://localhost/resource/request/0/1>;

	<http://unifiedviews.eu/ontology/dpu/httpPost/argument> <http://localhost/resource/request/0/2>;

	<http://unifiedviews.eu/ontology/dpu/httpPost/argument> <http://localhost/resource/request/0/3>.



<http://localhost/resource/request/0/1>

    <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <http://unifiedviews.eu/ontology/dpu/httpPost/Argument>;

    <http://unifiedviews.eu/ontology/dpu/httpPost/key> "type";

	<http://unifiedviews.eu/ontology/dpu/httpPost/value> "data_bodu".



<http://localhost/resource/request/0/2>

    <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <http://unifiedviews.eu/ontology/dpu/httpPost/Argument>;

    <http://unifiedviews.eu/ontology/dpu/httpPost/key> "mapa";

	<http://unifiedviews.eu/ontology/dpu/httpPost/value> "E01".



<http://localhost/resource/request/0/3>

    <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <http://unifiedviews.eu/ontology/dpu/httpPost/Argument>;

    <http://unifiedviews.eu/ontology/dpu/httpPost/key> "obdobi";

	<http://unifiedviews.eu/ontology/dpu/httpPost/value> "R2015".
